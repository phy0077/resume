% A ressection is a main section (<H1>Section</H1>)
\newenvironment{ressection}[1]{
	\vspace{4pt}
	{\fontfamily{phv}\selectfont\Large#1} \\
	\vspace{-8pt} \rule{\textwidth}{1pt}
	\vspace{1pt}
	\begin{itemize}
}{
	\end{itemize}	
}

% A ressection is a main section (<H1>SectionNo</H1>)
\newenvironment{ressectionno}[1]{
	\vspace{4pt}
	{\fontfamily{phv}\selectfont\Large#1} \\
	\vspace{-8pt} \rule{\textwidth}{1pt}
	\vspace{1pt}
}{	
}

% A resitem is a simple list element in a ressection (first level)
\newcommand{\resitem}[1]{
	\vspace{-4pt}
	\item \begin{flushleft} #1 \end{flushleft}
}

% A ressubitem is a simple list element in anything but a ressection (second level)
\newcommand{\ressubitem}[1]{
	\vspace{-1pt}
	\item \begin{flushleft} \large{#1} \end{flushleft}
}

% A resbigitem is a complex list element for stuff like jobs and education:
%  Arg 1: Name of company or university
%  Arg 2: Location
%  Arg 3: Title and/or date range
\newcommand{\resbigitem}[4]{
	\vspace{-5pt}
	\textbf{#1}---\large#2 
	\itshape\hfill\textit{\large#4}
	\newline\vspace{3pt}{\Large{#3}}
}

% This is a list that comes with a resbigitem
\newenvironment{ressubsec}[5]{
	\resbigitem{#1}{#2}{#3}{#4}
	\vspace{-1pt}
	\newline\vspace{3pt}{#5}
	\begin{itemize}
}{
	\end{itemize}
}

% This is a list that comes with a resbigitem
\newenvironment{ressubsecno}[4]{
        \newline
	\resbigitem{#1}{#2}{#3}{#4}
	\vspace{-1pt}
	\newline
}{
}

% This is a simple sublist
\newenvironment{reslist}[1]{
	\resitem{\textbf{#1}}
	\vspace{-5pt}
	\begin{itemize}
}{
	\end{itemize}
}
